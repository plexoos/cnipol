96  UInt_t countsR = 0;
97

99  {

101  countsR += hDetVsBunchId_ss.GetBinContent(iBunch, iDetPair->second);
102  }
103 
104  ValErrPair bunchAsym = CalcAsym(countsL, countsR, totalL, totalR);
