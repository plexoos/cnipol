1052  TF1* funcPCPolar = grPCPolar->GetFunction("fitFunc");
1053  if (!funcPCPolar) {
1054  Warning("CalcAvrgPolar", "Fill %d. No fit function found in graph for polId %d", fFillId, polId);
1055  return avrgPol;

1059  ERingId ringId = RunConfig::GetRingId(polId);
1060

1064  avrgPol = CalcAvrgPolarUnweighted(polId);
1065  return avrgPol;
1066  }
1067

1073

1075  {

1077  polar = funcPCPolar->Eval(time);
1078

1080  denom += intens;

1084  numerErr += polarErr*intens;
1085  denomErr += intens;
1086  }
1087 
1088  // Internaly all polarization values are between 0 and 1 except graphs
1089  avrgPol.first = numer/denom/100.;
