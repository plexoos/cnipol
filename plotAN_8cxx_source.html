259 void f_AN(double cal_t[], double cal_AN[], double rho, double sigma, double Imr5, double Rer5, double scale)
260 {
261  int i;
262  double mb2GeV_inv2 = 0.389379; // 1GeV-2=0.389mb factor to convert mb to GeV
263  double beta = 0;
264  double delta = 0.02;
265  double delta_t = ((5E-2) - (1E-4)) / (double)256;
266 
267  double sigma_totP = sigma / mb2GeV_inv2;
268 
269  double tc = -8 * TMath::Pi() * ALPHA / (sigma_totP * sqrt(1 + rho * rho));
270 
271  for (i = 0; i < 256; ++i) {
272  cal_t[i] = 0.5E-3 + delta_t * i;
273 
274  double tt = -1 * tc / cal_t[i];
275  double ttt = tt * tt - 2 * (rho + delta) * tt + (1 + rho * rho) * (1 + beta * beta);
276  double ans = tt * ((MAGMOM_PROTON - 1) * (1 - delta * rho) - 2 * (Imr5 - delta * Rer5)) - 2 * (Rer5 - rho * Imr5);
277 
278  cal_AN[i] = scale * ans * sqrt(cal_t[i]) / (double)(ttt * MASS_PROTON);
279  }
280 }

284 Double_t modelAN(Double_t *x, Double_t *par)
285 {
286  double minus_t = x[0];
287 
288  double Rer5 = par[0];
289  double Imr5 = par[1];
290 
291  double mb2GeV_inv2 = 0.389379; // 1GeV-2 = 0.389mb
292 
293  // sqrt(s) = 21.9 GeV, 255 GeV beam on H-jet target
294  double B = 12;
295  double rho = -0.0123;
296  double sigma_totP = 39.245;
297 
298  double Lambda2 = 0.71; //GeV^2
299  double gamma = 0.5772156;
300  double kappa = MAGMOM_PROTON - 1; // anomalous magnetic moment of proton
301 
302  sigma_totP /= mb2GeV_inv2;
303 
304  double delta = ALPHA*TMath::Log( 2./ minus_t / (B + 8./Lambda2) ) - ALPHA*gamma ;
305  double tc = -8 * TMath::Pi() * ALPHA / sigma_totP ;
306  double tt = -1 * tc / minus_t;
307  double denom = tt * tt - 2 * (rho + delta) * tt + (1 + rho * rho); // * (1 + beta * beta);
308  double nom = tt * (kappa * (1 - delta * rho) + 2 * (delta * Rer5 - Imr5)) - 2 * (Rer5 - rho * Imr5);
309 
310  return TMath::Sqrt(minus_t) * nom / MASS_PROTON / denom;
311 }
