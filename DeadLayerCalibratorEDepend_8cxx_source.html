296 Double_t BananaFitFunctor::operator()(double *x, double *p)
297 {
298  Double_t pp[5];
299  Double_t x0 = x[0];
300  Double_t x2 = x0*x0;
301 
302  Double_t p1 = TMath::Abs(p[1]);
303  Double_t p2 = p1*p1;
304  Double_t p3 = p1*p1*p1;
305 
306  pp[0] = cp0[0] + cp0[1]*p1 + cp0[2]*p2 + cp0[3]*p3;
307  pp[1] = cp1[0] + cp1[1]*p1 + cp1[2]*p2 + cp1[3]*p3;
308  pp[2] = cp2[0] + cp2[1]*p1 + cp2[2]*p2 + cp2[3]*p3;
309  pp[3] = cp3[0] + cp3[1]*p1 + cp3[2]*p2 + cp3[3]*p3;
310  pp[4] = cp4[0] + cp4[1]*p1 + cp4[2]*p2 + cp4[3]*p3;
311 
312  Double_t Ekin = pp[0] + pp[1]*x0 + pp[2]*x2 + pp[3]*x2*x0 + pp[4]*x2*x2;
313 
314  Double_t t_meas = (Ekin != 0.0) ? fRunConst.E2T/sqrt(Ekin) - p[0] : 0.0;
315 
316  return t_meas;
