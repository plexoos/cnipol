325 void AsymAnaInfo::VerifyOptions()
326 {
327  // The run name must be specified
328  if (fRunName.empty()) {
329  Error("VerifyOptions", "Run name has to be specified");
330  PrintUsage();
331  exit(EXIT_FAILURE);
332  }
333 
334  fOutputName = fRunName;
335  gMeasInfo->SetRunName(fRunName);
336 
337  // Check whether the raw data file exists
338  TString fileName = fRunName + ".data";
339 
340  if ( !gSystem->FindFile(fAsymEnv["CNIPOL_DATA_DIR"].c_str(), fileName ) )
341  {
342  Error("VerifyOptions", "Raw data file \"%s.data\" not found in %s\n", fRunName.c_str(), fAsymEnv["CNIPOL_DATA_DIR"].c_str());
343  exit(EXIT_FAILURE);
344  }
345 
346  if (HasAlphaBit()) {
347  fAlphaCalibRun = fRunName;
348  gMeasInfo->SetMeasType(kMEASTYPE_ALPHA);
349  }
350 
351  // Disable channels if requested by user
352  if ( fDisabledDetectors.any() ) {
353  gMeasInfo->DisableChannels(fDisabledDetectors);
354  }
355  printf("Checking right before AnaInfo::VerifyOptions()\n");
356  // The output dir is created here
357  AnaInfo::VerifyOptions();
358  printf("Checking right after AnaInfo::VerifyOptions()\n");
359  fFileRunConf = fopen(GetRunConfFileName().c_str(), "w");
360  gSystem->Chmod(GetRunConfFileName().c_str(), 0775);
361 }
