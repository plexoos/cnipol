472 void AnaGlobResult::CalcPolarNorm()
473 {
474  Info("CalcPolarNorm()", "Called");
475 
476  // Create containers for polarization where both hjet and pcarbon results exist
477  PolId2ValErrSet polarCrbSet;
478  PolId2ValErrSet polarJetSet;
479 
480  // Containers for polar vs prof polar normalization
481  PolId2ValErrSet polarCrbAllSet;
482  PolId2ValErrSet polarCrbProfAllSet;
483 
484 
485  AnaFillResultMapIter iFill = fAnaFillResults.begin();
486 
487  for ( ; iFill != fAnaFillResults.end(); ++iFill)
488  {
489  UInt_t fillId = iFill->first;
490  AnaFillResult *fillRslt = &iFill->second;
491 
492  PolId2ValErrMapConstIter iPolar = fillRslt->fPCPolars.begin();
493  PolId2ValErrMapConstIter iProfPolar = fillRslt->fPCProfPolars.begin();
494 
495  for ( ; iPolar != fillRslt->fPCPolars.end(); ++iPolar, ++iProfPolar)
496  {
497  EPolarimeterId polId = iPolar->first;
498  ValErrPair polarPC = iPolar->second;
499  ValErrPair polarPCProf = iProfPolar->second;
500  ERingId ringId = RunConfig::GetRingId(polId);
501  string sPolId = RunConfig::AsString(polId);
502 
503  // Polarization profile scale factor
504  if (polarPC.second >= 0 && polarPCProf.second >= 0) {
505  polarCrbAllSet [polId].insert(polarPC);
506  polarCrbProfAllSet[polId].insert(polarPCProf);
507 
508  fNormProfPolar2[polId] = utils::CalcWeightedAvrgErr(fNormProfPolar2[polId], utils::CalcDivision(polarPCProf, polarPC, 1));
509  }
510 
511  ValErrPair polarHJ = fillRslt->GetHJPolar(ringId);
512 
513  // For HJet normalization: Skip if there is no corresponding hjet result for this fill
514  if (polarPC.second < 0 || polarHJ.second < 0) {
515  Warning("CalcPolarNorm", "Fill %d, %s: p-Carbon or H-jet polarization not available", fillId, sPolId.c_str());
516  continue;
517  }
518 
519  TargetUId targetUId = fillRslt->GetPCTarget(polId);
520 
521  // The fills in which the carbon target is not well defined (multiple
522  // targets used in a fill) are not used in the normalization
523  // The idea behind this is to be consistent with scale factors by target
524  if (!targetUId.IsValid()) continue;
525 
526  polarCrbSet[polId].insert(polarPC);
527  polarJetSet[polId].insert(polarHJ);
528 
529  ValErrPair norm = utils::CalcDivision(polarHJ, polarPC);
530  fNormJetCarbon2[polId] = utils::CalcWeightedAvrgErr(fNormJetCarbon2[polId], norm);
531 
532  fNormJetCarbonByTarget2[targetUId] = utils::CalcWeightedAvrgErr(fNormJetCarbonByTarget2[targetUId], norm);
533  }
534  }
535 
536  // Now calculate the weighted average for each polarimeter
537  PolarimeterIdSetIter iPolId = gRunConfig.fPolarimeters.begin();
538 
539  for ( ; iPolId != gRunConfig.fPolarimeters.end(); ++iPolId)
540  {
541  if ( polarCrbSet.find(*iPolId) != polarCrbSet.end() ) {
542  ValErrPair avrgCrb = utils::CalcWeightedAvrgErr(polarCrbSet[*iPolId]);
543  ValErrPair avrgJet = utils::CalcWeightedAvrgErr(polarJetSet[*iPolId]);
544 
545  fNormJetCarbon[*iPolId] = utils::CalcDivision(avrgJet, avrgCrb);
546  }
547 
548  if ( polarCrbAllSet.find(*iPolId) != polarCrbAllSet.end() ) {
549  ValErrPair avrgPolar = utils::CalcWeightedAvrgErr(polarCrbAllSet[*iPolId]);
550  ValErrPair avrgProfPolar = utils::CalcWeightedAvrgErr(polarCrbProfAllSet[*iPolId]);
551 
552  fNormProfPolar[*iPolId] = utils::CalcDivision(avrgProfPolar, avrgPolar);
553  }
554  }
555 }
