136  UsePlainAlphaGain();
137  return;
138  }
139

141  iCh != fChannelCalibs.end(); iCh++) {
142  int chId = iCh->first;
143  if (chId == 0)
144  {
145  continue; // skip channel containing mean data
146  }
147

152
